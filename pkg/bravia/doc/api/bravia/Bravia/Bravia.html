Bravia(String host,
    {int port = 80, String psk = '0000', int timeout = 5000}) {
  if (!host.contains('.')) {
    host += '.';
  }
  _host = host;
  _port = port;
  _psk = psk;
  _timeout = timeout;

  _url = _port != 80 ? 'http://$_host:$_port/sony' : 'http://$_host/sony';

  debug('Construct Bravia $host $psk $timeout $_url');

}
